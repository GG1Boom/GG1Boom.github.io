#基本标签

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>基本标签联练习</title>
</head>
<body>

<!--标题标签-->
<h1>一级标签</h1>
<h2>二级标签</h2>
<h3>三级标签</h3>
<h4>四级标签</h4>
<h5>五级标签</h5>
<h6>六级标签</h6>

<!--段落标签-->
<p>两只老虎</p>
<p>跑得快</p>

<!--换行标签-->
两只老虎<br/>
跑得快<br/>

<!--水平线标签-->
<hr>

<!--粗体，斜体-->
<h1>字体样式标签</h1>

粗体：<strong>I Love U</strong>
斜体：<em>I Love U</em>
<br/>
<!--特殊符号-->
空     格：&nbsp
<br/>
空     &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;   格：
<br/>
大于号： >
<br/>
小于号： <
<br/>
&copy 版权所有阿川
</body>
</html>

表单标签

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>登录注册</title>
</head>

<!--form
action: 处理表单的位置
method: post | get
get会显示在url上显示密码
-->

<body>
<h1>登录</h1>
    <form action="01我的第一个网页.html" method="get">

<!--文本输入框
input type="text" 123456
input type="password" *******

type: text,password,checkbox,radio,submit,reset,file,hidden,image,button，image
name:
value: 指定一个初始值，当type为radio或checkbox时必须指定一个
size: 表单元素的初始大小
maxlength: text和password的最大长度
checked: type为radio或checkbox时，指定按钮是否被选中

-->
    <p>名字：<input type="text" name="username" value="achuan" size="30" maxlength="7" placeholder="请输入用户名" required></p>
    <p><input type="password" name="pwd" hidden value="123456"></p>
    <p>
        <input type="submit">
        <input type="reset" value="achuan">
    </p>

    <p>
        <input type="radio" value="boy" name="sex" id="boy" checked><label for="boy">男</label>
        <input type="radio" value="girl" name="sex" id="girl"><label for="girl">女</label>
    </p>
<!--增强鼠标可用性-->
        <p>
            <label for="ser">搜索</label>
        </p>
<!--多选框
input type="checkbox" value="sleep" name="hobby"
input type="button" name="btn1" value="点击变长"
input type="image" src="../resources/image/1.JPG"
-->
    <p>爱好：
        <input type="checkbox" value="sleep" name="hobby">睡觉
        <input type="checkbox" value="coding" name="hobby" checked>敲代码
        <input type="checkbox" value="eating"  name="hobby" disabled>干饭
        <input type="checkbox" value="fighting"  name="hobby">干架
        <input type="checkbox" value="gaming"  name="hobby">打游戏
    </p>

    <p>按钮：
        <input type="button" name="btn1" value="点击变长">
        <input type="image" src="../resources/image/1.JPG" width="300" height="200">
    </p>

    <p>地址：
        <select name="国家">
            <option value="cn">中国</option>
            <option value="japan">日本</option>
            <option value="kr" selected>韩国</option>
            <option value="un">美国</option>
        </select>
        <select name="地区">
            <option value="sc">四川</option>
            <option value="gz">广州</option>
            <option value="sz">深圳</option>
        </select>
    </p>
    <p>文本域：
        <textarea name="textarea" cols="30" rows="10" readonly>文本</textarea>
    </p>

    <p>文件域
        <input type="file" name="files">
        <input type="button" value="上传" name="upload">

    </p>
    <p>邮箱：
        <input type="email" name="email">

    </p>
    <p>
        自定义标签：
        <input type="text" name="diymail" pattern="^\w+([-+.]\w+)*@\w+([-.]\w+)*\.\w+([-.]\w+)*$">
    </p>
    <p>url：
        <input type="url" name="url">
    </p>

    <p>数字：
        <input type="number" name="number" max="1000" step="1" min="0">
    </p>

    <p>滑块
        <input type="range" name="voice" max="100" min="50" step="2">
    </p>

    <p>搜索
        <input type="search" name="search" id="ser">

</form>
